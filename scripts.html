import os
from datetime import datetime
import clipboard

def get_time(epoch=False):
  os.system("date +%s | pbcopy")
  time_epoch = int(clipboard.paste())
  time_in_proper_format = datetime.fromtimestamp(time_epoch).strftime("%A, %B %d, %Y %I:%M:%S")
  if epoch:
    return time_epoch
  return [time_epoch, time_in_proper_format]
